# Denmark Flag - Assembly (EMU8086)

## 📌 คำอธิบายโปรเจค
โปรเจคนี้เป็นการเขียนโปรแกรมภาษา Assembly เพื่อวาด "ธงชาติเดนมาร์ก" โดยใช้โปรแกรม EMU8086  
แสดงผลในโหมดกราฟิก (Mode 13h) ขนาด 320x200 พิกเซล

---

## ⚙️ หลักการทำงาน
- ใช้ interrupt `int 10h` เพื่อเข้าสู่โหมดกราฟิก
- เขียนข้อมูลลง Video Memory (A000h) โดยตรง
- ใช้คำสั่ง `rep stosw` เพื่อเพิ่มความเร็วในการวาดภาพ
- วาดพื้นหลังสีแดง และวาดกากบาทสีขาวตามรูปแบบธงเดนมาร์ก

## 🧾 Source Code

org 100h

; 1. เข้าสู่โหมดกราฟิก (VGA 320x200, 256 สี)
mov ax, 13h
int 10h
mov ax, 0A000h
mov es, ax

; ==========================================
; ส่วนที่ 1: วาดพื้นหลัง 3 แถบแนวนอน (เขียว, ขาว, แดง)
; ==========================================

; แถบสีเขียว (บนสุด) - ความสูง 67 บรรทัด
mov di, 0
mov cx, 10720    ; คำนวณจาก: (67 บรรทัด * 320 พิกเซล) / 2
mov ax, 0202h    ; ใช้รหัสสีเขียว (02) เติมทีละ 2 พิกเซล
rep stosw

; แถบสีขาว (ตรงกลาง) - ความสูง 67 บรรทัด
mov cx, 10720    
mov ax, 0F0Fh    ; ใช้รหัสสีขาว (15) เติมทีละ 2 พิกเซล
rep stosw

; แถบสีแดง (ล่างสุด) - ความสูง 66 บรรทัด
mov cx, 10560    ; คำนวณจาก: (66 บรรทัด * 320 พิกเซล) / 2
mov ax, 0404h    ; ใช้รหัสสีแดง (04) เติมทีละ 2 พิกเซล
rep stosw

; ==========================================
; ส่วนที่ 2: วาดสี่เหลี่ยมคางหมูสีดำด้านซ้าย
; (แบ่งการวาดเป็น 3 ส่วน เพื่อไล่ระดับความชันของมุม)
; ==========================================

; ส่วนย่อยที่ 1: มุมเฉียงลง (ทับบนแถบสีเขียว)
mov di, 0        ; เริ่มต้นที่มุมซ้ายบนสุด (พิกัด X=0, Y=0)
mov cx, 67       ; จำนวนบรรทัดที่ต้องวาด (67 บรรทัด)
mov bx, 0        ; ตั้งค่าความกว้างเริ่มต้นของสีดำ = 0
mov bp, 0        ; ตัวแปรสำหรับเก็บค่าเศษส่วนความชัน
top_trap:
    push di      ; บันทึกตำแหน่งเริ่มต้นของบรรทัดปัจจุบัน
    push cx
    mov cx, bx   ; กำหนดความกว้างของสีดำในบรรทัดนี้
    jcxz skip_t  ; ถ้าความกว้างเป็น 0 ให้ข้ามการวาดไป
    mov al, 0    ; รหัสสีดำ (0)
    rep stosb    ; ระบายสีดำลงไป
skip_t:
    pop cx
    pop di
    add di, 320  ; ขยับตัวชี้ตำแหน่งไปยังบรรทัดถัดไป
    
    ; คำนวณเพื่อเพิ่มความกว้างของสีดำ (เฉียงออกไปทางขวา)
    add bp, 80
calc_t:
    cmp bp, 67
    jl  next_t
    inc bx       ; ขยายความกว้างขึ้นทีละ 1 พิกเซล
    sub bp, 67
    jmp calc_t
next_t:
    loop top_trap

; ส่วนย่อยที่ 2: สี่เหลี่ยมตรงกลาง (ทับบนแถบสีขาว)
; ณ จุดนี้ แถบสีดำจะมีความกว้าง 80 พิกเซลพอดี
mov cx, 67       ; จำนวนบรรทัดที่ต้องวาด (67 บรรทัด)
mid_trap:
    push di
    push cx
    mov cx, 80   ; รักษาระดับความกว้างคงที่ 80 พิกเซล
    mov al, 0    ; รหัสสีดำ
    rep stosb
    pop cx
    pop di
    add di, 320  ; ขยับตัวชี้ตำแหน่งไปยังบรรทัดถัดไป
    loop mid_trap

; ส่วนย่อยที่ 3: มุมเฉียงกลับเข้ามา (ทับบนแถบสีแดง)
mov cx, 66       ; จำนวนบรรทัดที่เหลือ (66 บรรทัด)
mov bx, 80       ; เริ่มต้นที่ความกว้าง 80 พิกเซล
mov bp, 0
bot_trap:
    push di
    push cx
    mov cx, bx   ; กำหนดความกว้างตามค่าในตัวแปร bx
    jcxz skip_b
    mov al, 0
    rep stosb
skip_b:
    pop cx
    pop di
    add di, 320  ; ขยับตัวชี้ตำแหน่งไปยังบรรทัดถัดไป
    
    ; คำนวณเพื่อลดความกว้างของสีดำ (เฉียงกลับเข้าหาขอบซ้าย)
    add bp, 80
calc_b:
    cmp bp, 66
    jl  next_b
    dec bx       ; หดความกว้างลงทีละ 1 พิกเซล
    sub bp, 66
    jmp calc_b
next_b:
    loop bot_trap

; ==========================================
; ส่วนที่ 3: จบการทำงานและคืนค่าระบบ
; ==========================================

; หยุดรอรับคำสั่ง (รอผู้ใช้กดปุ่มใดๆ บนคีย์บอร์ด)
mov ah, 00h
int 16h

; ล้างหน้าจอและกลับเข้าสู่โหมดตัวอักษร (Text Mode) ตามปกติ
mov ax, 03h
int 10h

; จบโปรแกรมและคืนหน่วยความจำให้ระบบ (Exit to DOS)
mov ah, 4Ch
int 21h
